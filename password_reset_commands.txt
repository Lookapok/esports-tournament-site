# 重設密碼指令

# 方法 1: 使用 Django Shell
python manage.py shell

# 在 Shell 中執行：
from django.contrib.auth.models import User
user = User.objects.get(username='dkka')
user.set_password('你的新密碼')
user.save()
print('密碼已更新')
exit()

# 方法 2: 一行指令
python manage.py shell -c "
from django.contrib.auth.models import User; 
user = User.objects.get(username='dkka'); 
user.set_password('admin123'); 
user.save(); 
print('✅ 密碼已重設為: admin123')
"

# 方法 3: 使用 changepassword 命令
python manage.py changepassword dkka

# 登入資訊
# 用戶名: dkka
# 新密碼: admin123
# 登入網址: http://127.0.0.1:8000/login/
